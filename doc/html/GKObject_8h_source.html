932  {

934

936  qInfo( "GKObject__idComparer error: comparing NULL objects" );

938  res = true;
939  }
940  }else if( s1 != s2 && s1->getId() == s2->getId() ){
941  qInfo( "GKObject__idComparer error: two objects with the same id %p %p %i", s1, s2, s1->getId() );

944  if( s1->getId() <= 0 && s2->getId() <= 0 && s1 != s2 ){
945  qInfo( "GKObject__idComparer error: objects with invalid IDs %p %i %p %i", s1, s1->getId(), s2, s2->getId() );
946  }else if( s1->getId() <= 0 ){
947  qInfo( "GKObject__idComparer error: object with invalid ID %p %i", s1, s1->getId() );

949  qInfo( "GKObject__idComparer error: object with invalid ID %p %i", s2, s2->getId() );
